   135Рєњ   135Рєњ      (window.navigator as Navigator & { standalone?: boolean }).standalone === true;

   140Рєњ   140Рєњ      setShowButton(true);
   141Рєњ   141Рєњ      return;
   142Рєњ   142Рєњ    }
   143Рєњ   143Рєњ

   145Рєњ   145Рєњ    const handler = (e: Event) => {

   163Рєњ   163Рєњ      setDeferredPrompt(null);

   165Рєњ   165Рєњ  }, [deferredPrompt, isIOS]);
